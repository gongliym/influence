def load_fdict(filename, encoding='gb18030'):
    fdict = {}
    for line in codecs.open(filename, 'r', encoding):
        fid, fname = line.strip().split()
        fdict[int(fid)] = fname
    return fdict

def load_fnames(filename, encoding = 'gb18030'):
    fnames = []
    for line in codecs.open(filename, 'r', encoding):
        fid, fname = line.strip().split()
        fnames.append(fname)
    return fnames

def translate(inst, fnames, featIdx):
    feats = []
    for each_idx in featIdx:
        fval = inst[each_idx]
        fname = fnames[each_idx]
        feats.append('%s:%s' % (fname, fval))
    return feats

def cal_auc(pred_res):
    list_fpr, list_tpr = [], []
    true_pos, false_pos = 0., 0.
    num_pos = len([1 for (p,l) in pred_res if l==1])
    num_neg = len([1 for (p,l) in pred_res if l==0])

    for (score, label) in sorted(pred_res, key=lambda x:x[0], reverse=True):
        if label == 1 : true_pos += 1
        else : false_pos += 1
        tpr = true_pos / num_pos
        fpr = false_pos / num_neg
        list_fpr.append(fpr)
        list_tpr.append(tpr)

    auc_score, prev_fpr = 0., 0.
    for fpr, tpr in zip(list_fpr, list_tpr):
        if fpr != prev_fpr:
            auc_score += (fpr - prev_fpr) * tpr
            prev_fpr = fpr
    return auc_score

def getInstanceOfInterest(predictions, labels):
    assert len(predictions) == len(labels)
    pos_inst = {}
    neg_inst = {}
    for idx, (pred, label) in enumerate(zip(predictions, labels)):
        assert label == 1 or label == 0
        if label == 1: pos_inst[idx] = pred
        else: neg_inst[idx] = pred
    return pos_inst, neg_inst
